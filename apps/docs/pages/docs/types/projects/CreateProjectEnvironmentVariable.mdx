export interface CreateProjectEnvironmentVariable {
  /**
   * The git branch of the environment variable
   */
  gitBranch?: string;

  /**
   * Name of the ENV variable
   */

  type?: 'system' | 'secret' | 'encrypted' | 'plain';

  /**

}
